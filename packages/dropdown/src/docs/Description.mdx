import { Container, Row, Col } from 'storybook/blocks';
import Dropdown from '../Dropdown';

## Default

You can add label and options(children)

```tsx live
render(
  <Container>
    <Row flex align="top">
      <Col span="24">
        <Dropdown id="marque" labelFor="marque" label="Marque label">
          <option value="Marque">Marque</option>
          <option value="saab">Saab</option>
          <option value="opel">Opel</option>
          <option value="audi">Audi</option>
        </Dropdown>
      </Col>
    </Row>
  </Container>,
);
```

## Disabled

The `disabled` attribute is a boolean attribute it makes the dropdown/select element is unusable and un-clickable.

```tsx live
render(
  <Container>
    <Row flex align="top">
      <Col span="24">
        <Dropdown
          id="marque"
          disabled={true}
          labelFor="marque"
          label="Marque label"
        >
          <option value="Marque">Marque</option>
          <option value="saab">Saab</option>
          <option value="opel">Opel</option>
          <option value="audi">Audi</option>
        </Dropdown>
      </Col>
    </Row>
  </Container>,
);
```

## Error

The `error` prop toggles the error state and can then be used to provide feedback to the user about the `error`.

```tsx live
render(
  <Container>
    <Row flex align="top">
      <Col span="24">
        <Dropdown
          id="marque"
          error="This is an error"
          labelFor="marque"
          label="Marque label"
        >
          <option value="Marque">Marque</option>
          <option value="saab">Saab</option>
          <option value="opel">Opel</option>
          <option value="audi">Audi</option>
        </Dropdown>
      </Col>
    </Row>
  </Container>,
);
```
