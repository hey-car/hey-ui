import { Container, Row, Col } from 'storybook/blocks';
import Dropdown from '../Dropdown';
import { AC } from '../../../icons/src';

## Default

Use the `options` prop to to pass options.

```tsx live
<Container>
  <Row
    style={{ width: '100%', height: '200px', justifyContent: 'space-between' }}
  >
    <Col span={12}>
      <Dropdown
        onChange={() => {}}
        options={[
          {
            value: 'pomelo',
            label: 'Pomelo',
          },
          {
            value: 'apple',
            label: 'Apple',
          },
          {
            value: 'mango',
            label: 'Mango',
          },
        ]}
        value={{
          value: 'mango',
          label: 'Mango',
        }}
      ></Dropdown>
    </Col>
  </Row>
</Container>
```
