import { Container } from 'storybook/blocks';
import Pagination from '../Pagination';

## Basic usage

Controlling the `Pagination` component with onClick event
```tsx live
render(
  <Container>

<Row
    align="middle"
    style={{ width: '100%', justifyContent: 'space-between' }}
  >
    <Col span={3}>
       <Pagination
          totalPages={10}
          currentPage={3}
          onClick={itemNumber => console.log(itemNumber)}
       />
    </Col>
  </Row>

  </Container>,
);
```

## Using hrefs
```tsx live
render(
   <Pagination
      totalPages={10}
      currentPage={1}
      renderItem={item => (
          <PaginationItem
            href={`https://hey.car/page/${item.page}`}
            {...item}
          />
        )}
   />
);
```

## Custom aria-labels
```tsx live
render(
   <Pagination
      totalPages={10}
      currentPage={3}
      aria-label={'Paginação'}
      getItemAriaLabel={({ type, page, selected }) => {
        return 'Vá para a página ' + page;
      }}
   />
);
```

## Custom component
```tsx
interface LinkProps {
  link: string;
}
const Link: React.FC<LinkProps> = ({ link, children, ...rest }) => (
  <a href={link} {...rest}>
    {children}
  </a>
);

render(
   <Pagination
      totalPages={3}
      currentPage={3}
      renderItem={item => (
          <PaginationItem
            component={Link}
            link={`https://hey.car/page/${item.page}`}
            {...item}
          />
        )}
   />
);
```

## Accessibility

`Pagination` supports keyboard navigation and is compatible with screen readers.
Pressing the `Tab` and `Shift + Tab` keys will move focus between buttons.
Pressing `Enter` will do a click event
