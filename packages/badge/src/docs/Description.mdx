import { Container, Row, Col } from 'storybook/blocks';
import Badge from '../Badge';
import { Lorry } from '../../../icons/src';

## Variants

Use the `color` prop to change the visual style of the Badge.
You can set the value to `primary`, `secondary` or `tertiary`.

```tsx live
const badgeVariants = ['primary', 'secondary', 'tertiary'];

render(
  <Container>
    <Row align="middle">
      {badgeVariants.map(variant => (
        <Col key={variant}>
          <Badge leftIcon={<Lorry fontSize="18" />} color={variant}>
            $250 off
          </Badge>
        </Col>
      ))}
    </Row>
  </Container>,
);
```
