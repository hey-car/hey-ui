import { Container, Row, Col } from 'storybook/blocks';
import Button from '../Button';

## Variants of the button

```tsx live
const buttonVariants = ['contained', 'outlined', 'link'];

render(
  <Container>
    <Row align="large">
      {buttonVariants.map(variant => (
        <Col key={variant}>
          <Button size="large" variant={variant}>
            Button CTA
          </Button>
        </Col>
      ))}
    </Row>
  </Container>,
);
```

## Sizes of the button

```tsx live
const buttonSizes = ['large', 'small'];

render(
  <Container>
    <Row style={{ width: '100%' }}>
      {buttonSizes.map(size => (
        <Col span="2" key={size}>
          <Button size={size} variant="contained">
            Button CTA
          </Button>
        </Col>
      ))}
    </Row>
    <Row style={{ width: '100%' }}>
      {buttonSizes.map(size => (
        <Col span="2" key={size}>
          <Button size={size} color="primary" variant="outlined">
            Button CTA
          </Button>
        </Col>
      ))}
    </Row>
    <Row style={{ width: '100%' }}>
      {buttonSizes.map(size => (
        <Col span="2" key={size}>
          <Button size={size} variant="link">
            Button CTA
          </Button>
        </Col>
      ))}
    </Row>
  </Container>,
);
```
